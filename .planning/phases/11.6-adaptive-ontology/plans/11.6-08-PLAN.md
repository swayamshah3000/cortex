```typescript
export function useProminentEntities(limit: number = 5) {
  return useQuery({
    queryKey: queryKeys.prominentEntities,
    queryFn: async () => {
      // Fetch top entities across all types, sort by documentCount desc.
      const [persons, orgs, locations] = await Promise.all([
        tauriInvoke<CanonicalEntity[]>("get_entities_by_type", { entityType: "person" }, () => []),
        tauriInvoke<CanonicalEntity[]>("get_entities_by_type", { entityType: "organization" }, () => []),
        tauriInvoke<CanonicalEntity[]>("get_entities_by_type", { entityType: "location" }, () => []),
      ]);
      const all = [...persons, ...orgs, ...locations];
      const sorted = all.sort((a, b) => b.documentCount - a.documentCount).slice(0, limit);
      return sorted.map((e, i) => ({ entity: e, docCount: e.documentCount, isProminent: i < 3 }));
    },
  });
}
```

```typescript
/**
 * Returns the top-N entities across Person/Organization/Location by documentCount.
 * Used by Sidebar top-5 prominence (Plan 09 D-13). Query is derived client-side
 * from existing get_entities_by_type IPC.
 */
export function useProminentEntities(limit: number = 5) {
  return useQuery({
    queryKey: [...queryKeys.prominentEntities, limit],
    queryFn: async () => {
      const [persons, orgs, locations] = await Promise.all([
        tauriInvoke<CanonicalEntity[]>("get_entities_by_type", { entityType: "person" }, () => mockProminentEntities.filter(p => p.entity.entityType === "person").map(p => p.entity)),
        tauriInvoke<CanonicalEntity[]>("get_entities_by_type", { entityType: "organization" }, () => mockProminentEntities.filter(p => p.entity.entityType === "organization").map(p => p.entity)),
        tauriInvoke<CanonicalEntity[]>("get_entities_by_type", { entityType: "location" }, () => mockProminentEntities.filter(p => p.entity.entityType === "location").map(p => p.entity)),
      ]);
      const all = [...persons, ...orgs, ...locations];
      const sorted = all.sort((a, b) => b.documentCount - a.documentCount).slice(0, limit);
      return sorted.map((entity, i) => ({
        entity,
        docCount: entity.documentCount,
        isProminent: i < 3,
      })) as ProminentEntity[];
    },
  });
}
```
