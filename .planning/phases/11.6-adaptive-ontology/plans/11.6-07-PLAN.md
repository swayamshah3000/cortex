**Struct + run function:**

```rust
pub struct ConsolidationLoop {
    auth: Arc<AuthState>,
    ontology_store: Arc<Mutex<OntologyStore>>,
    app_data_dir: std::path::PathBuf,
}

impl ConsolidationLoop {
    pub fn new(auth, ontology_store, app_data_dir) -> Self;

    /// Single-pass: check triggers, call LLM if due, store proposals. Never applies.
    /// Returns Ok(true) if a consolidation ran, Ok(false) if skipped.
    pub async fn run_once(&self) -> Result<bool, AppError> {
        // Gate 1: opt-in check.
        let (enabled, triples_since, last_at) = {
            let os = self.ontology_store.lock().await;
            (os.automatic_growth_enabled(),
             os.triples_since_last_consolidation(),
             os.last_consolidation_at())
        };
        if !enabled {
            return Ok(false);
        }

        // Gate 2: trigger check.
        let hours_since = last_at.and_then(|s| chrono::DateTime::parse_from_rfc3339(&s).ok())
            .map(|dt| chrono::Utc::now().signed_duration_since(dt).num_hours() as u64)
            .unwrap_or(u64::MAX); // never consolidated → always trigger

        let due = triples_since >= CONSOLIDATION_TRIGGER_TRIPLES ||
                  hours_since >= CONSOLIDATION_TRIGGER_HOURS;
        if !due {
            return Ok(false);
        }

        // Gate 3: provider check.
        let provider = match self.auth.get_active_provider().map_err(AppError::Internal)? {
            Some(p) => p,
            None => return Ok(false),
        };

        // Read snapshot of vocabulary.
        let (predicates, subclasses) = {
            let os = self.ontology_store.lock().await;
            (os.effective_predicates(), os.entity_subclasses())
        };

        // Build user content.
        let vocab_lines: String = predicates.iter()
            .filter(|p| p.source != PromotionSource::Seed) // Rule 2: don't propose seed changes
            .map(|p| format!("- {} ({}): {}", p.name, format!("{:?}", p.source).to_lowercase(), p.description))
            .collect::<Vec<_>>().join("\n");
        let subclass_lines: String = subclasses.iter()
            .map(|s| format!("- {}.{}: {}", s.class, s.subclass, s.example_value.as_deref().unwrap_or("")))
            .collect::<Vec<_>>().join("\n");
        let user_content = format!(
            "Current vocabulary (non-seed only):\n{}\n\nEntity subclasses:\n{}\n\nAnalyze and propose consolidations per the rules above.",
            vocab_lines, subclass_lines
        );

        // Skip if vocabulary is too small for consolidation to be useful.
        if predicates.iter().filter(|p| p.source != PromotionSource::Seed).count() < 3 {
            let mut os = self.ontology_store.lock().await;
            os.reset_consolidation_counter(&chrono::Utc::now().to_rfc3339());
            return Ok(false);
        }

        // AIServiceRequest — reuse Pass 2 model selection.
        let req = AIServiceRequest {
            system_prompt: CONSOLIDATION_PROMPT.to_string(),
            messages: vec![ServiceMessage { role: "user".into(), content: user_content }],
            max_tokens: Some(2048),
            temperature: Some(0.2),
            response_format: None,
            model_override: None,
        };
        let response = ai_request_with_retry(self.auth.as_ref(), req, 3).await
            .map_err(|e| AppError::Internal(format!("Consolidation LLM call failed: {}", e)))?;

        // Parse.
        let raw = parse_consolidation_json(&response.content)?;
        let now = chrono::Utc::now().to_rfc3339();
        let pending = validate_consolidation(raw, &predicates, &crate::types::SEED_PREDICATES, now.clone(), response.model_used.clone(), predicates.len() as u32);

        // Store.
        {
            let mut os = self.ontology_store.lock().await;
            os.set_pending_consolidation(Some(pending));
            os.reset_consolidation_counter(&now);
        }

        // Persist.
        {
            let os_arc = self.ontology_store.clone();
            let dir = self.app_data_dir.clone();
            tokio::task::spawn_blocking(move || {
                let os = os_arc.blocking_lock();
                let _ = os.save(&dir);
            }).await.ok();
        }

        Ok(true)
    }
}
```

**Public spawner:**

```rust
/// Spawn a background tokio task that runs `run_once` every 60 seconds.
/// Backpressure: tokio interval ticks yield; a slow LLM call blocks the next tick until it returns.
/// Never crashes the runtime — errors are logged and swallowed.
/// Returns immediately; the task lives for the app lifetime.
pub fn spawn_consolidation_task(
    auth: Arc<AuthState>,
    ontology_store: Arc<Mutex<OntologyStore>>,
    app_data_dir: std::path::PathBuf,
) {
    tauri::async_runtime::spawn(async move {
        let cl = ConsolidationLoop::new(auth, ontology_store, app_data_dir);
        let mut interval = tokio::time::interval(tokio::time::Duration::from_secs(60));
        interval.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Delay);
        loop {
            interval.tick().await;
            match cl.run_once().await {
                Ok(true) => eprintln!("[consolidation] proposals stored (pending user approval)"),
                Ok(false) => { /* silent — either not due, opt-out, or vocab too small */ }
                Err(e) => eprintln!("[consolidation] error: {} (retrying next tick)", e),
            }
        }
    });
}
```
