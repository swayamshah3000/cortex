```typescript
export function useBackfillProgress() {
  useEffect(() => {
    if (!isTauri()) return;
    let unlisten: (() => void) | undefined;
    (async () => {
      const { listen } = await import("@tauri-apps/api/event");
      unlisten = await listen<EntityBackfillProgress>("entity-backfill-progress", (event) => {

      });
    })();
    return () => { unlisten?.(); };
  }, []);
}
```

```typescript
export function useSpaceLabels() {
  return useQuery({
    queryKey: queryKeys.spaceLabels,
    queryFn: () => tauriInvoke<Record<string, SpaceLabelEntry>>("get_space_labels", {}, () => ({})),
  });
}

export function useRenameSpace() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: ({ spaceId, newLabel }: { spaceId: string; newLabel: string }) =>
      tauriInvoke<SpaceLabelEntry>(
        "rename_space_label",
        { spaceId, newLabel },
        () => ({ fingerprint: "mock", label: newLabel, description: "", canonicalEntityHint: undefined, generatedAt: new Date().toISOString(), userLocked: true }),
      ),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: queryKeys.spaces });
      queryClient.invalidateQueries({ queryKey: queryKeys.spaceLabels });
    },
  });
}

export function useClearSpaceOverride() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: (spaceId: string) =>
      tauriInvoke<SpaceLabelEntry>(
        "clear_space_override",
        { spaceId },
        () => ({ fingerprint: "mock", label: "", description: "", canonicalEntityHint: undefined, generatedAt: new Date().toISOString(), userLocked: false }),
      ),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: queryKeys.spaces });
      queryClient.invalidateQueries({ queryKey: queryKeys.spaceLabels });
    },
  });
}

export function useRelabelSpace() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: (spaceId: string) =>
      tauriInvoke<Space>("trigger_relabel", { spaceId }, () => {
        throw new Error("Cannot relabel in mock runtime");
      }),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: queryKeys.spaces });
      queryClient.invalidateQueries({ queryKey: queryKeys.spaceLabels });
    },
  });
}
```

IPC argument names MUST be camelCase (`spaceId`, `newLabel`) — Tauri v2 serde converts these to snake_case Rust params (`space_id`, `new_label`). Verify against existing hooks (e.g. `move_document_to_space` uses `docId` / `spaceId`).
