```tsx
import { useNavigate } from "react-router-dom";
import { ContextMenu, ContextMenuTrigger, ContextMenuContent, ContextMenuItem } from "@/components/ui/context-menu";
import { openPath, revealItemInDir } from "@tauri-apps/plugin-opener";
import { FileText, ExternalLink, FolderOpen } from "lucide-react";
import { toast } from "sonner";
import { isTauri } from "@/lib/tauri";
import type { Document } from "@/lib/types";

export function DocumentContextMenu({ doc, children }: { doc: Document; children: React.ReactNode }) {
  const navigate = useNavigate();
  const handleOpen = () => navigate(`/document/${doc.id}`);
  const handleOpenExternal = async () => {
    if (!isTauri()) return;
    try { await openPath(doc.path); }
    catch { toast.error("Could not open file. Open it manually from the file manager."); }
  };
  const handleReveal = async () => {
    if (!isTauri()) return;
    try { await revealItemInDir(doc.path); }
    catch { toast.error("Could not reveal file in Finder."); }
  };
  return (
    <ContextMenu>
      <ContextMenuTrigger asChild>{children}</ContextMenuTrigger>
      <ContextMenuContent>
        <ContextMenuItem onClick={handleOpen}><FileText className="mr-2 h-4 w-4" /> Open</ContextMenuItem>
        <ContextMenuItem onClick={handleOpenExternal}><ExternalLink className="mr-2 h-4 w-4" /> Open in default app</ContextMenuItem>
        <ContextMenuItem onClick={handleReveal}><FolderOpen className="mr-2 h-4 w-4" /> Reveal in Finder</ContextMenuItem>
      </ContextMenuContent>
    </ContextMenu>
  );
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User → folder picker | User chooses any directory accessible by the app process; path returned is fed into addFolder (which calls add_watched_folder IPC) |
| User → context menu | User right-clicks; openPath / revealItemInDir receive doc.path values from query results (server-side validated paths from watched-folders ingestion) |
| Frontend → Tauri plugins | All native I/O routes through plugin-dialog and plugin-opener, which have their own capability checks (Plan 01) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-FOLDER-INVALID | Tampering / Validation | User selects a path that no longer exists or stopped being a directory by the time addFolder fires (race) | mitigate | Per D-19: client-side `@tauri-apps/plugin-fs::exists` + `stat().isDirectory` validation BEFORE the addWatchedFolder IPC. Failed validation shows toast "Selected path is no longer a directory" and aborts. Backend addWatchedFolder IPC re-validates server-side as defense in depth. Try/catch wraps the entire open+validate+submit flow. |
| T-06-FOLDER-SYMLINK | Information Disclosure | User picks a symlinked folder pointing outside their intent | accept | Native folder picker is the user's explicit choice; Cortex indexes whatever they pick. No new attack vs existing manual-path entry. |
| T-06-OPEN-PATH | Elevation of Privilege | openPath delivers any string to OS shell handler | mitigate | doc.path comes from server-side document metadata (indexed by Cortex from user-watched folders); no untrusted-input source. Server-side validation in T-06-OPN at Plan 01 establishes the trust boundary. |
| T-06-CTX-NAV | Spoofing | Open in context menu navigates to /document/{id} — id comes from result list | mitigate | id is a server-issued doc_id from the query response; not directly user-supplied to the context-menu handler. |
| T-06-BROWSER-FALLBACK | Information Disclosure | In browser dev mode, plugin calls would throw "no plugin" errors — could leak Tauri internals via console | mitigate | isTauri() guard at the top of every handler returns early; openPath/revealItemInDir never invoked from browser. WatchedPage's Add button is disabled in browser. |
</threat_model>
