```rust
pub struct PkceCodes {
    pub verifier: String,   // 43+ chars, URL-safe base64 no-pad
    pub challenge: String,  // SHA-256(verifier), URL-safe base64 no-pad
}

pub struct OAuthFlowConfig<'a> {
    pub provider_slug: &'a str,     // "openai-codex" | "gemini-oauth"
    pub authorization_url: &'a str, // e.g. "https://auth.openai.com/oauth/authorize"
    pub token_url: &'a str,         // e.g. "https://auth.openai.com/oauth/token"
    pub client_id: &'a str,
    pub scope: &'a str,             // space-delimited scope string
    pub port_range: (u16, u16),     // e.g. (1455, 1465) for codex, or (8000, 8010) for gemini
    pub redirect_path: &'a str,     // e.g. "/auth/callback" for codex, "/" for gemini
    pub redirect_uri_host: &'a str, // "localhost" for codex, "127.0.0.1" for gemini
                                    // ↑ CRITICAL per 07-OAUTH-RESEARCH.md §Redirect URI Host Resolution
                                    // Loopback socket bind is ALWAYS 127.0.0.1 in loopback.rs; this field controls
                                    // the redirect_uri STRING sent in the authorization URL and token exchange.
    pub extra_authz_params: &'a [(&'a str, &'a str)], // e.g. codex extras
    pub token_request_style: TokenRequestStyle,       // FormUrlencoded (codex) | Json (some providers)
}

pub enum TokenRequestStyle { FormUrlencoded, Json }

pub struct OAuthTokenResponse {
    pub access_token: String,
    pub refresh_token: Option<String>,
    pub expires_in: Option<i64>,   // seconds until expiry, from provider
    pub token_type: Option<String>,
}

pub async fn generate_pkce_codes() -> PkceCodes;
pub async fn start_oauth_flow(
    config: OAuthFlowConfig<'_>,
    on_authorization_url_ready: impl FnOnce(String) -> Result<(), String>,
) -> Result<OAuthTokenResponse, String>;

// Convenience for plan 07-09:
pub async fn refresh_access_token(
    refresh_url: &str,
    client_id: &str,
    refresh_token: &str,
    style: TokenRequestStyle,
) -> Result<OAuthTokenResponse, String>;
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| system browser → loopback listener | Browser is untrusted; state param must be constant-time compared to expected_state to prevent code-substitution attacks |
| provider → refresh endpoint | Provider is trusted for cryptographic verification but its response body may contain a leaked bearer token — do not log responses |
| loopback listener → local machine | Any local process could connect to the loopback port; state param + short 5-min timeout mitigate |
