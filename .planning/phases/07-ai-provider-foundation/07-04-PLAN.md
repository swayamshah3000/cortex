<objective>
Build the frontend foundation layer: TypeScript types mirroring the Rust IPC structs (Plan 03 outputs), 6 React Query hooks for invoking the AI provider IPC commands, and a session-only Zustand store for the "Connect AI" banner dismissed state. After this plan: any React component can call `useProviders()` to subscribe to the provider list, `useConnectProvider()` to register a provider, or read `useAiBannerStore().isDismissed` to gate the banner. Plans 05 and 06 build UI on top of these primitives — they don't need to know about Tauri invoke directly.

Purpose: Hooks and types are the seam between Tauri IPC and React rendering. Centralizing them in `useTauri.ts` (the existing convention from Phase 4) lets Plans 05 and 06 run in parallel — each consumes the same hooks without coupling. The `useAiBannerStore` must explicitly avoid `persist` middleware (D-15: banner returns on every launch until a provider is connected); the explicit test in `stores.test.ts` enforces this invariant.

Output: Modifications to 4 existing client files and 1 new test file. No new components (Plans 05/06 build those). No changes to backend or any other file.
</objective>

    ```typescript
    import { describe, it, expect } from "vitest";
    import { useAiBannerStore } from "./stores";

    describe("useAiBannerStore", () => {
      it("starts with isDismissed = false", () => {
        const initial = useAiBannerStore.getState();
        expect(initial.isDismissed).toBe(false);
      });

      it("dismiss() flips isDismissed to true", () => {
        useAiBannerStore.getState().dismiss();
        expect(useAiBannerStore.getState().isDismissed).toBe(true);
      });

      it("does not auto-rehydrate dismissed state across new store consumer mounts", () => {
        // Zustand persist middleware rehydrates state on store creation from storage.
        // Without persist, programmatically resetting state (simulating app reload)
        // returns to the initial value.
        useAiBannerStore.getState().dismiss();
        expect(useAiBannerStore.getState().isDismissed).toBe(true);

        // Simulate "app reload" by resetting state to initial. With persist, the dismissed
        // value would be re-loaded from storage. Without persist, our explicit reset wins.
        useAiBannerStore.setState({ isDismissed: false });
        expect(useAiBannerStore.getState().isDismissed).toBe(false);

        // Confirm Zustand exposes no `persist` API surface on this store (no `.persist` key
        // on the StoreApi — only stores wrapped with the persist middleware expose this).
        const api = useAiBannerStore as unknown as { persist?: unknown };
        expect(api.persist).toBeUndefined();
      });
    });
    ```

       ```typescript
       // === Phase 7: AI Provider Hooks ===

       export function useProviders() {
         return useQuery({
           queryKey: queryKeys.providers,
           queryFn: () => tauriInvoke<ProviderAuthStatus[]>("list_providers", {}, () => mockProviders),
           staleTime: 30_000,
         });
       }

       export function useConnectProvider() {
         const queryClient = useQueryClient();
         return useMutation({
           mutationFn: (request: ConnectProviderRequest) =>
             tauriInvoke<ProviderAuthStatus>("connect_provider", { request }, () => mockProviders[0]),
           onSuccess: () => {
             queryClient.invalidateQueries({ queryKey: queryKeys.providers });
             queryClient.invalidateQueries({ queryKey: queryKeys.activeProvider });
           },
         });
       }

       export function useDisconnectProvider() {
         const queryClient = useQueryClient();
         return useMutation({
           mutationFn: (provider: string) =>
             tauriInvoke<void>("disconnect_provider", { provider }, () => undefined as void),
           onSuccess: () => {
             queryClient.invalidateQueries({ queryKey: queryKeys.providers });
             queryClient.invalidateQueries({ queryKey: queryKeys.activeProvider });
           },
         });
       }

       export function useSetActiveProvider() {
         const queryClient = useQueryClient();
         return useMutation({
           mutationFn: (provider: string) =>
             tauriInvoke<void>("set_active_provider", { provider }, () => undefined as void),
           onSuccess: () => {
             queryClient.invalidateQueries({ queryKey: queryKeys.providers });
             queryClient.invalidateQueries({ queryKey: queryKeys.activeProvider });
           },
         });
       }

       export function useSaveSetupToken() {
         const queryClient = useQueryClient();

           mutationFn: (token: string) =>
             tauriInvoke<OAuthStartResult>("save_setup_token", { token }, () => ({ started: true, provider: "anthropic" })),
           onSuccess: () => {
             queryClient.invalidateQueries({ queryKey: queryKeys.providers });
           },
         });
       }

       export function useTestConnection() {
         return useMutation({
           mutationFn: (provider: string) =>
             tauriInvoke<void>("test_connection", { provider }, () => undefined as void),
         });
       }
       ```
