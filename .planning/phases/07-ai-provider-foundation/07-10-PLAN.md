// client/hooks/useTauri.ts

export function useStartOpenAiOAuth() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: () =>
      tauriInvoke<ProviderAuthStatus>("start_openai_oauth", {}, () => mockProviders[1]), // OpenAI slot mock
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: queryKeys.providers });
      queryClient.invalidateQueries({ queryKey: queryKeys.activeProvider });
    },
  });
}
