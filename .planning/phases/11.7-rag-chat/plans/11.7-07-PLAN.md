```tsx
{/* Chat Sessions Section — Phase 11.7 D-18 (RAGCH-05, RAGCH-06)
 *
 * Renders a "Chat" nav link + a drawer of the 5 most-recently-updated sessions.
 * The nav link always navigates to /chat (fresh session UI).
 * Each session row is a <Link to={/chat/{id}}> with an inline trash button
 * revealed on hover — same pattern as Saved Searches.
 * Collapsed sidebar: only the MessageSquare icon is shown.
 */}
<div className="pt-4">
  {!isCollapsed && (
    <div className="px-3 py-2 flex items-center justify-between">
      <span className="text-xs font-semibold uppercase tracking-wider text-text-tertiary">
        Chat
      </span>
    </div>
  )}

  {/* Primary "Chat" nav link */}
  <div className="space-y-1">
    <NavLink path="/chat" label="Chat" icon={MessageSquare} />
  </div>

  {/* Recent sessions (top 5 by updatedAt desc). Hidden when collapsed. */}
  {!isCollapsed && chatSessions && chatSessions.length > 0 && (
    <nav className="space-y-0.5 px-2 pt-1">
      {[...chatSessions]
        .sort((a, b) => b.updatedAt.localeCompare(a.updatedAt))
        .slice(0, 5)
        .map((session) => {
          const url = `/chat/${session.id}`;
          const active = isActive(url);
          return (
            <div key={session.id} className="group flex items-center gap-1 rounded-md hover:bg-bg-tertiary">
              <Link
                to={url}
                className={cn(
                  "flex flex-1 items-center gap-2 rounded-md px-3 py-1.5 text-xs font-medium transition-colors truncate",
                  active
                    ? "bg-bg-tertiary text-text-primary"
                    : "text-text-secondary hover:text-text-primary",
                )}
              >
                <span className="flex-1 truncate">{session.title}</span>
              </Link>
              <button
                type="button"
                aria-label="Delete chat"
                onClick={(e) => {
                  e.preventDefault();
                  e.stopPropagation();
                  if (window.confirm(`Delete chat "${session.title}"? This cannot be undone.`)) {
                    deleteChatSession.mutate(session.id);
                  }
                }}
                className="p-1 rounded hover:bg-bg-primary text-text-tertiary hover:text-red-400 transition-colors opacity-0 group-hover:opacity-100"
                style={{ minWidth: 32, minHeight: 32 }}
              >
                <Trash2 size={12} />
              </button>
            </div>
          );
        })}
      {chatSessions.length > 5 && (
        <Link
          to="/chat"
          className="block px-3 py-1.5 text-xs text-accent-primary hover:text-accent-hover transition-colors"
        >
          View all ({chatSessions.length})
        </Link>
      )}
    </nav>
  )}
</div>
```
