```typescript
/**
 * useChatStream — subscribes to chat-stream-token / -complete / -error events
 * for a specific (sessionId, assistantMessageId) pair. Accumulates tokens into
 * a local `assistantContent` state so the ChatPage can render live.
 *
 * Unmounting or changing target ids automatically unsubscribes.
 *
 * Returns:
 *   assistantContent: current accumulated tokens (streaming or final)
 *   citations: populated on 'complete'
 *   isStreaming: true after listeners attach, false after complete/error
 *   error: message string when 'error' fires
 */

import { useEffect, useState } from "react";
import { isTauri } from "@/lib/tauri";
import type {
  ChatStreamTokenEvent,
  ChatStreamCompleteEvent,
  ChatStreamErrorEvent,
  Citation,
} from "@/lib/types";

interface UseChatStreamArgs {
  sessionId: string | null;
  assistantMessageId: string | null;
}

interface UseChatStreamReturn {
  assistantContent: string;
  citations: Citation[];
  isStreaming: boolean;
  error: string | null;
  reset: () => void;
}

export function useChatStream({ sessionId, assistantMessageId }: UseChatStreamArgs): UseChatStreamReturn {
  const [assistantContent, setAssistantContent] = useState("");
  const [citations, setCitations] = useState<Citation[]>([]);
  const [isStreaming, setIsStreaming] = useState(false);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    if (!isTauri() || !sessionId || !assistantMessageId) return;
    setAssistantContent("");
    setCitations([]);
    setError(null);
    setIsStreaming(true);

    let unlistenToken: (() => void) | undefined;
    let unlistenComplete: (() => void) | undefined;
    let unlistenError: (() => void) | undefined;

    (async () => {
      const { listen } = await import("@tauri-apps/api/event");

      unlistenToken = await listen<ChatStreamTokenEvent>("chat-stream-token", (e) => {
        if (e.payload.sessionId !== sessionId || e.payload.messageId !== assistantMessageId) return;
        setAssistantContent((prev) => prev + e.payload.token);
      });

      unlistenComplete = await listen<ChatStreamCompleteEvent>("chat-stream-complete", (e) => {
        if (e.payload.sessionId !== sessionId || e.payload.messageId !== assistantMessageId) return;
        setCitations(e.payload.citations ?? []);
        setIsStreaming(false);
      });

      unlistenError = await listen<ChatStreamErrorEvent>("chat-stream-error", (e) => {
        if (e.payload.sessionId !== sessionId || e.payload.messageId !== assistantMessageId) return;
        setError(e.payload.error);
        setIsStreaming(false);
      });
    })();

    return () => {
      unlistenToken?.();
      unlistenComplete?.();
      unlistenError?.();
    };
  }, [sessionId, assistantMessageId]);

  const reset = () => {
    setAssistantContent("");
    setCitations([]);
    setError(null);
    setIsStreaming(false);
  };

  return { assistantContent, citations, isStreaming, error, reset };
}
```

**Create `client/pages/ChatPage.tsx`:**

```tsx
import { useEffect, useState, useMemo } from "react";
import { useParams, useNavigate, Link } from "react-router-dom";
import { Plus } from "lucide-react";
import { toast } from "sonner";
import { ChatMessageList } from "@/components/chat/ChatMessageList";
import { ChatInput } from "@/components/chat/ChatInput";
import { useChatSessions, useStartChat } from "@/hooks/useTauri";
import { useChatStream } from "@/hooks/useChatStream";
import type { ChatMessage } from "@/lib/types";

export default function ChatPage() {
  const { sessionId: routeSessionId } = useParams<{ sessionId?: string }>();
  const navigate = useNavigate();
  const { data: sessions, refetch: refetchSessions } = useChatSessions();
  const startChat = useStartChat();

  // Local id triple set on submit; cleared on new-chat.
  const [activeSessionId, setActiveSessionId] = useState<string | null>(routeSessionId ?? null);
  const [assistantMessageId, setAssistantMessageId] = useState<string | null>(null);

  useEffect(() => {
    setActiveSessionId(routeSessionId ?? null);
    setAssistantMessageId(null);
  }, [routeSessionId]);

  const currentSession = useMemo(
    () => sessions?.find((s) => s.id === activeSessionId) ?? null,
    [sessions, activeSessionId],
  );

  const { assistantContent, citations, isStreaming, error, reset } = useChatStream({
    sessionId: activeSessionId,
    assistantMessageId,
  });

  // On stream complete: refetch sessions so the persisted assistant message + citations
  // hydrate into the message list (replacing the transient streaming placeholder).
  useEffect(() => {
    if (!isStreaming && assistantMessageId) {
      refetchSessions();
    }
  }, [isStreaming, assistantMessageId, refetchSessions]);

  useEffect(() => {
    if (error) toast.error(error);
  }, [error]);

  const handleSubmit = async (query: string) => {
    try {
      const result = await startChat.mutateAsync({
        query,
        sessionId: activeSessionId,
        filters: null,   // TODO Plan 07 follow-up: pass current filter context if user chats from a filtered view (D-04)
      });
      setActiveSessionId(result.sessionId);
      setAssistantMessageId(result.assistantMessageId);
      if (!routeSessionId) {
        navigate(`/chat/${result.sessionId}`, { replace: true });
      }
    } catch (e) {
      toast.error(e instanceof Error ? e.message : String(e));
    }
  };

  const handleNewChat = () => {
    reset();
    setActiveSessionId(null);
    setAssistantMessageId(null);
    navigate("/chat");
  };

  const messages: ChatMessage[] = currentSession?.messages ?? [];

  // Deduplicate: if the streaming message id already exists in currentSession.messages
  // (i.e. refetchSessions completed), do NOT render the streaming placeholder.
  const hasPersistedAssistant = assistantMessageId != null &&
    messages.some((m) => m.id === assistantMessageId);
  const streamingMessage = isStreaming || (!hasPersistedAssistant && assistantContent.length > 0)
    ? { id: assistantMessageId ?? "streaming", content: assistantContent, citations }
    : null;

  return (
    <div className="flex flex-col h-[calc(100vh-120px)]">
      <div className="flex items-center justify-between px-4 py-3 border-b border-border-primary">
        <h1 className="text-lg font-semibold text-text-primary">
          {currentSession?.title ?? "New Chat"}
        </h1>
        <button
          onClick={handleNewChat}
          className="inline-flex items-center gap-1.5 rounded-md border border-border-primary bg-bg-secondary px-3 py-1.5 text-sm font-medium text-text-secondary hover:bg-bg-tertiary hover:text-text-primary transition-colors"
        >
          <Plus size={14} />
          New Chat
        </button>
      </div>

      <ChatMessageList messages={messages} streamingMessage={streamingMessage} />

      <div className="border-t border-border-primary bg-bg-primary p-4">
        <ChatInput onSubmit={handleSubmit} disabled={isStreaming || startChat.isPending} />
      </div>
    </div>
  );
}
```
